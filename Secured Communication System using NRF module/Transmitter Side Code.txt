const char* ssid     = "YOUR_WIFI_NAME";

//RF address (must match receiver) 
const byte address[6] = "SECR1";

//AES Key & IV
byte aesKey[16] = {
  0x2B, 0x7E, 0x15, 0x16, 0x28, 0xAE, 0xD2, 0xA6,
  0xAB, 0xF7, 0x15, 0x88, 0x09, 0xCF, 0x4F, 0x3C
};
byte aesIV[16] = {
  0x00, 0x01, 0x02, 0x03, 0x04, 0x05, 0x06, 0x07,
  0x08, 0x09, 0x0A, 0x0B, 0x0C, 0x0D, 0x0E, 0x0F
};

char    inputMsg[17] = "";
uint8_t encBuf[32];
bool    newMessage = false;
String  lastStatus = "No message sent yet.";

//Handle root page
void handleRoot() {
  server.send(200, "text/html", getPage());
}

//Handle message send 
void handleSend() {
  if (server.hasArg("msg")) {
    String msg = server.arg("msg");
    msg.trim();

    if (msg.length() == 0) {
      lastStatus = "Empty message!";
    } else {
      if (msg.length() > 16) msg = msg.substring(0, 16);

      memset(inputMsg, 0, sizeof(inputMsg));
      memcpy(inputMsg, msg.c_str(), msg.length());
      newMessage = true;

      Serial.print(">> Web message: \"");
      Serial.print(inputMsg);
      Serial.println("\"");
    }
  }
  server.sendHeader("Location", "/");
  server.send(303);
}







void setup() {
  Serial.begin(115200);
  delay(1000);

  //WiFi
  Serial.print("Connecting to WiFi: ");
  Serial.println(ssid);
  WiFi.begin(ssid, password);

  int tries = 0;
  while (WiFi.status() != WL_CONNECTED && tries < 20) {
    delay(500);
    Serial.print(".");
    tries++;
  }

  if (WiFi.status() == WL_CONNECTED) {
    Serial.println("\nWiFi Connected!");
    Serial.print("Open this IP in your browser: http://");
    Serial.println(WiFi.localIP());
  } else {
    Serial.println("\nWiFi FAILED â€” check credentials.");
  }

  //Web Server
  server.on("/",     HTTP_GET,  handleRoot);
  server.on("/send", HTTP_POST, handleSend);
  server.begin();
  Serial.println("Web server started.");

  //Radio
  SPI.begin(18, 19, 23, 5);

  if (!radio.begin()) {
    Serial.println("ERROR: nRF24L01 not found!");
    while (1);
  }

  delay(100);
  radio.setPALevel(RF24_PA_MIN);
  radio.setDataRate(RF24_1MBPS);
  radio.setChannel(76);
  radio.setCRCLength(RF24_CRC_16);
  radio.setAutoAck(true);
  radio.openWritingPipe(address);
  radio.stopListening();

  aesLib.set_paddingmode(paddingMode::CMS);

  Serial.println("Radio OK! System ready.");
}






void loop() {
  server.handleClient();  // handle web requests

  if (newMessage) {
    byte iv[16];
    memcpy(iv, aesIV, 16);
    memset(encBuf, 0, sizeof(encBuf));
    aesLib.encrypt((byte*)inputMsg, 16, encBuf, aesKey, 128, iv);

    bool success = radio.write(encBuf, 32);

    if (success) {
      Serial.println("Transmitted successfully!");
      lastStatus = " Sent: " + String(inputMsg);
    } else {
      Serial.println("Transmission FAILED.");
      lastStatus = " Failed to send!";
    }

    newMessage = false;
  }

  delay(10);
}
